        for(i=0;i<10;i++)

            for(j=i+1;j<10;j++)
            {
                if(ar[i]<ar[j]) //desc order so that the highest no come in the 0th index
                {

                    ar[i]=ar[j];
                    ar[j]=temp;
                }
            }
        }
         printf("\n");

        }
