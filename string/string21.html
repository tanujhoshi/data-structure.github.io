#include <stdio.h>
            #include<string.h>
            void permute(char str[], int currentIndex, int length) {
                if (currentIndex == length) {
                    printf("%s\n", str);
                    return;
                }
            
                for (int i = currentIndex; i <= length; i++) {
                    // Swap characters
                    char temp = str[currentIndex];
                    str[currentIndex] = str[i];
                    str[i] = temp;
            
                    permute(str, currentIndex + 1, length);
            
                    temp = str[currentIndex];
                    str[currentIndex] = str[i];
                    str[i] = temp;
                }
            }
            
            void main() {
                char str[100];
                printf("Enter a string: ");
                scanf("%s", str);
                int n = strlen(str);
                printf("Permutations of the string are:\n");
                permute(str, 0, n - 1);
            }
