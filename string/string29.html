                void stringToAsciiArray(char str[], int asciiArray[]) {
                    int length = strlen(str);
                
                    for (int i = 0; i < length; i++) {
                        asciiArray[i] = (int)str[i];
                    }
                }
                
                int main() {
                    char str[100]; // Assuming a string of up to 100 characters
                    int asciiArray[100]; // Assuming the same length for ASCII values
                
                    printf("Enter a string: ");
                    gets(str); // Read a line of text (be cautious about buffer overflow)
                
                    // Convert the string to an array of ASCII values
                    stringToAsciiArray(str, asciiArray);
                
                    printf("ASCII values of the characters in the string:\n");
                    for (int i = 0; i < strlen(str); i++) {
                        printf("%c: %d\n", str[i], asciiArray[i]);
                    }
