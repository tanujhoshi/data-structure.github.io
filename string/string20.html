#include <stdio.h>
        #include <string.h>
            
        void main()
            {
                char str[100];
                int countchar, countwords, countspace, countlines;
                int i;
                countchar = 0;
                countwords = 0;
                countspace = 0;
                countlines = 0;
                printf("Enter a string : ");
                gets(str);
                for (i = 0; str[i]; i++)
                {
                    if ((str[i] >= 65 && str[i] <= 90) || (str[i] >= 97 && str[i] <= 122))
                    {
                        countchar = countchar + 1;
                    }
                    else if (str[i] == ' ' || str[i] == '.')
                    {
                        if (str[i] == ' ')
                        {
                            countspace = countspace + 1;
                        }
                        countwords = countwords + 1;
                    }
                }
                if (str[i - 1] != '.')
                {
                    countwords = countwords + 1;
                }
                printf("number of character = %d", countchar);
                printf("number of spaces = %d", countspace);
                printf("number of words = %d", countwords);
                printf("number of lines = 1");
            }
