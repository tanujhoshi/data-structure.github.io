            //find the smallest element in the integer array using recursion
            int min(int arr[], int n)
            {
                if (n == 1) 
                {
                    return arr[0];
                } else 
                    {
                        if (arr[n-1] < min(arr, n - 1)) {
                            return arr[n-1];
                    } else 
                    {
                        return min(arr, n - 1);
                    }
                    }
            }
