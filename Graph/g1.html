#include <stdio.h>
                int main() {
                    int n, e, i, j, vertex1, vertex2;
                    printf("Enter the number of vertices:\n");
                    scanf("%d", &n);
            
                    int adjMatrix[n][n];
            
                    // Initialize the adjacency matrix with 0s.
                    for(i = 0; i < n; i++) {
                        for(j = 0; j < n; j++) {
                            adjMatrix[i][j] = 0;
                        }
                    }
            
                    printf("Enter the number of edges:\n");
                    scanf("%d", &e);
            
                    // Read edges and update the adjacency matrix.
                    for(i = 0; i < e; i++) {
                        printf("Enter edge (format: vertex1 vertex2):\n");
                        scanf("%d %d", &vertex1, &vertex2);
            
                        // Check for valid vertices and update the matrix.
                        if(vertex1 >= 0 && vertex1 < n && vertex2 >= 0 && vertex2 < n) {
                            adjMatrix[vertex1][vertex2] = 1;
                            adjMatrix[vertex2][vertex1] = 1; // For undirected graph. Remove this for directed graph.
                        } else {
                            printf("Invalid vertices. Please enter vertices between 0 to %d\n", n - 1);
                            i--; // to ensure the correct number of edges are inputted.
                        }
                    }
            
                    // Print the adjacency matrix.
                    printf("Adjacency Matrix:\n");
                    for(i = 0; i < n; i++) {
                        for(j = 0; j < n; j++) {
                            printf("%d ", adjMatrix[i][j]);
                        }
                        printf("\n");
                    }
            
                    return 0;
                }
