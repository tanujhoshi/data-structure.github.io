            // this will swap for us and will mak eprogram less complicated.
            void swap(char *one, char *two) {
                char temp[100];
                strcpy(temp, one);
                strcpy(one, two);
                strcpy(two, temp);
            }
            
            // Function to perform Bubble Sort
            void string_bubble(char *strng[], int n) {
                int i, j;
                for (i = 0; i < n - 1; i++) {
                    for (j = 0; j < n - i - 1; j++) {
                        if (strcmp(strng[j], strng[j + 1]) > 0) {
                            //we can also compare string manually by matching character to character 
                            swap(strng[j], strng[j + 1]);
                        }
                    }
                }
            }
            
            
            int main() {
                int i,n=0;
                printf("Enter the number of elements in the array: ");
                scanf("%d", &n);
            
                char *arr[n];//let's get this thing clear that this is the size of string array
                for (i = 0; i < n; i++) {
                    arr[i]=(char*)malloc(100); //and here memory is allocated to the string not array
                    printf("Enter string %d: ", i + 1);
                    scanf("%s", arr[i]);
                }
            
                //start time
                clock_t start = clock();
            
                // Perform Bubble Sort
                string_bubble(arr, n);
                //end time
                clock_t end = clock();
            
                printf("Sorted array using Bubble Sort:\n");
                for (i = 0; i < n; i++) {
                    printf("%s\n ", arr[i]);
                }
            
                // Calculate the time taken in seconds
                double timeTaken = ((double)(end - start)) / CLOCKS_PER_SEC;
                printf("\nTime taken: %f seconds\n", timeTaken);
            
                 // Deallocate memory for strings
                for (i = 0; i < n; i++) {
                    free(arr[i]);
                }
                return 0;
            }
