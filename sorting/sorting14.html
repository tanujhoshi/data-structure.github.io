#include <stdio.h>
            #include <string.h>
            #include <stdlib.h>
            #include <time.h>
            
            void insert_string_recur(char *arr[], int n) {
                if (n <= 1) return; //if the array has only one or zero elements it's already sorted.
            
                // Recursively sort the first n-1 elements
                //already told in last program about why we do n-1
                insert_string_recur(arr, n - 1);
            
                // Insert the last element into the sorted part of the array
                char *key = arr[n - 1];//set value
                int j = n - 2;//compare with predecessor value
                //same logic
                while (j >= 0 && strcmp(arr[j], key) > 0) {
                    arr[j + 1] = arr[j];
                    j--;
                }
            //assigning, already told in last program why j+1
                arr[j + 1] = key;
            }
            
            int main()
            {
                int i, n = 0;
                printf("Enter the number of elements in the array: ");
                scanf("%d", &n);
            
                char *arr[n]; // let's get this thing clear that this is the size of string array
                for (i = 0; i < n; i++)
                {
                    arr[i] = (char *)malloc(100); // and here memory is allocated to the string not array
                    printf("Enter string %d: ", i + 1);
                    scanf("%s", arr[i]);
                }
            
                // start time
                clock_t start = clock();
            
                // Perform Bubble Sort
                insert_string_recur(arr, n);
                // end time
                clock_t end = clock();
            
                printf("Sorted array using Inseetion Sort:\n");
                for (i = 0; i < n; i++)
                {
                    printf("%s\n ", arr[i]);
                }
            
                // Calculate the time taken in seconds
                double timeTaken = ((double)(end - start)) / CLOCKS_PER_SEC;
                printf("\nTime taken: %f seconds\n", timeTaken);
            
                // Deallocate memory for strings
                for (i = 0; i < n; i++)
                {
                    free(arr[i]);
                }
                return 0;
            }
