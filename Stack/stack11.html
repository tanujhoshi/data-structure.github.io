    int top=-1;

    void push(float stack[],int n,float value){
         if (top==n-1)
         printf("overflow condition ");
        else
           top++;
    
        stack[top]=value;

    float  pop(float stack[],int n){
        if (top==-1)
           printf("underflow ");
        else{
            val=stack[top];
    
           top--;}
        return val;
    }
