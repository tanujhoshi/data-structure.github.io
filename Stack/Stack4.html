#include <stdio.h>
               #include <stdlib.h>
                           
               struct stack {
                   char data;
                   struct stack* next;
               };
               
               typedef struct stack node;
               
               node* start = NULL, * nn;
               
               void push(char value) {
                   if (start == NULL) {
                       start = (node*)malloc(sizeof(node));
                       start->data = value;
                       start->next = NULL;
                       printf("\nValue has been inserted for the 1st time");
                       return;
                   }
                   else {
                       nn = (node*)malloc(sizeof(node));
                       if (nn != NULL) {
                           nn->data = value;
                           nn->next = start;
                           start = nn;
                           printf("\nValue has been inserted");
                       }
                       else {
                           printf("\nStack is already full....");
                       }
                   }
               }
               
               char pop() {
                   if (start == NULL) {
                       printf("\nStack is empty");
                       return '\0'; // Return a default value to indicate an empty stack
                   }
                   char value = start->data;
                   start = start->next;
                   printf("\nValue has been deleted.");
                   return value;
               }
               
               int isEmpty() {
                   if (start == NULL) {
                       printf("\nStack is empty");
                       return 1; // Return 1 to indicate an empty stack
                   }
                   else {
                       printf("\nStack is not empty yet");
                       return 0; // Return 0 to indicate a non-empty stack
                   }
               }
               
               void display() {
                   if (start == NULL) {
                       printf("\nStack is empty");
                       return;
                   }
                   node* items = start;
               
                   printf("Stack: ");
                   while (items != NULL) {
                       printf("%c -> ", items->data);
                       items = items->next;
                   }
                   printf("NULL\n");
               }
               
               int main() {
                   int ch;
                   char value;
               
                   printf("Press 1: Insert a character.\n");
                   printf("Press 2: Delete a character.\n");
                   printf("Press 3: Display stack.\n");
                   printf("Press 4: Check if empty.\n");
                   printf("Press 5: Exit the program.\n");
               
                   while (1) {
                       printf("\nEnter your choice: ");
                       scanf("%d", &ch);
               
                       switch (ch) {
                       case 1:
                           printf("Enter character to insert: ");
                           scanf(" %c", &value); 
                           push(value);
                           break;
                       case 2:
                           value = pop();
                           if (value != '\0') {
                               printf("Popped value: %c\n", value);
                           }
                           break;
                       case 3:
                           display();
                           break;
                       case 4:
                           if (isEmpty()) {
                               printf("\nStack is empty.\n");
                           }
                           else {
                               printf("\nStack is not empty.\n");
                           }
                           break;
                       case 5:
                           printf("\nCode has been terminated.\n");
                           exit(0);
                       default:
                           printf("\nInvalid Choice.\n");
                       }
                   }
               
                   return 0;
               }
