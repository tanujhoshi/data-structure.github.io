#include<stdio.h<
#include<math.h<
int top=-1;
float val;
void push(float stack[],int n,float value){
     if (top==n-1)
     printf("overflow condition ");
    else
       top++;

    stack[top]=value;
    
  
}
float  pop(float stack[],int n){
    if (top==-1)
       printf("underflow ");
    else{
        val=stack[top];

       top--;}
    return val;
}
int main()
{
char eq[20];
float stack[20];
float operator1,operator2,sol,num;
printf("Enter the equation:");
gets(eq);
for(int i=0;eq[i]!='\0';i++){
    if(isdigit(eq[i])){
num=eq[i]-48;
   push(stack,20,num);
    }
    else{
   operator1=pop(stack,20);
  
   operator2=pop(stack,20);
   
   if(eq[i]=='+')
   sol=operator2+operator1;
    else if(eq[i]=='-')
   sol=operator2-operator1;
    else if(eq[i]=='*')
   sol=operator2*operator1;
    else if(eq[i]=='/')
   sol=operator2/operator1;
    else if(eq[i]=='^')
   sol=pow(operator2,operator1);
   else
   printf("Operator is not identified\n");
    push(stack,20,sol);
    }
     
}
printf("Answer is %f",pop(stack,20));
