            void isEmpty(int i){
                if(i == -1){
                    printf("\nQueue is empty");
                }
                else{
                    printf("\nQueue is not empty yet");
                }
                
            }
            int peek(int arr[],int i){
                if(i == -1){
                    isEmpty(i); 
                    return -1;
                }
                return i;
            }
            
            void isFull(int size,int i){
                if(i == size-1){
                    printf("\nQueue is Fulll");
                    return;
                }
                printf("\nQueue is not yet Full");
            }
            int enqueue(int arr[],int size,int value,int i,int *j){
                if(i == size-1){
                    isFull(size,i);
                    return;
                }else{
                    if(*j==-1){
                        *j=0;
                    }
                    i++;
                    arr[i]=value;
                }
                 return i;
            }
            int dequeue(int arr[],int size,int *j){
                if(*j == -1){
                    isEmpty(*j);
                    return;
                }
                (*j)--;
                
                return *j;
            }
            void display(int arr[],int size,int i,int j){
                if (j == -1) {
                    isEmpty(j);
                }
                else{
                int k;
                for(k=j;k<=i;k++){
                    printf("%d ",arr[k]);
                }
                    
                }
            }
