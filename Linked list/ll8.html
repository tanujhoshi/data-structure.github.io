#include <stdio.h>
        #include <stdlib.h>
        
        typedef struct Node
        {
            char *data;
            struct Node *next;
            struct Node *prev;
        } Node;
        
        typedef struct LinkedList
        {
            Node *head;
            Node *end;
            int length;
        } LinkedList;
        
        LinkedList *init()
        {
            LinkedList *list = (LinkedList *)malloc(sizeof(LinkedList));
            if (list == NULL)
            {
                printf(" \nMemory allocation failed! \n");
                return list;
            }
        
            list->head = NULL;
            list->end = NULL;
            list->length = 0;
            return list;
        }
        
        int strlen(char str[])
        {
            int i;
            for (i = 0; str[i]; i++)
                ;
            return i;
        }
        
        void removeNewLineCharacter(char input[])
        {
            int length = strlen(input);
        
            if (length > 0 && input[length - 1] == '\n')
            {
                input[length - 1] = '\0';
            }
        }
        
        void print(LinkedList *list)
        {
            if (list->head == NULL)
                return;
        
            Node *temp = list->head;
            printf("\n       \nList: ");
            do
            {
                printf("%s -> ", temp->data);
                temp = temp->next;
            } while (temp != list->head);
        
            printf("\n       \n");
        }
        
        void insertAtBeg(LinkedList *list)
        {
        
            Node *newNode = (Node *)malloc(sizeof(Node));
            if (newNode == NULL)
            {
                printf(" \nList is Full \n");
                return;
            }
            getchar();
            char input[100];
            printf("Enter the value to insert: ");
            fgets(input, 100, stdin);
            removeNewLineCharacter(input);
            int length = strlen(input);
        
            if (length == 0)
            {
                printf(" **Value cannot be empty! **\n");
                return;
            }
        
            newNode->data = (char *)malloc(sizeof(char) * length);
        
            for (int i = 0; i <= length; i++) // i<=length because null character should also be included
            {
                newNode->data[i] = input[i];
            }
        
            if (list->head == NULL)
            {
                newNode->next = newNode;
                newNode->prev = newNode;
                list->end = newNode;
            }
            else
            {
                newNode->next = list->head;
                newNode->prev = list->end;
                list->end->next = newNode;
                list->head->prev = newNode;
            }
            list->head = newNode;
            (list->length)++;
        }
        
        void printBackward(LinkedList *list)
        {
            if (list->end == NULL)
                return;
        
            Node *temp = list->end;
            printf("\n       \nList: ");
        
            do
            {
                printf("%s -> ", temp->data);
                temp = temp->prev;
            } while (temp != list->end);
            printf("\n       \n");
        }

        void deleteAtBeg(LinkedList *list)
        {
            if (list->head == NULL)
                return;
        
            Node *nodeToDelete = list->head;
        
            if (list->length == 1) // only one node in the list
            {
                list->head = NULL;
                list->end = NULL;
            }
            else
            {
                list->head = nodeToDelete->next;
                nodeToDelete->next->prev = list->end;
                list->end->next = list->head;
            }
        
            free(nodeToDelete);
            (list->length)--;
        }
        
        void deleteAtEnd(LinkedList *list)
        {
            if (list->end == NULL)
            {
                return;
            }
        
            Node *nodeToDelete = list->end;
        
            if (list->length == 1) // only one node in the list
            {
                list->head = NULL;
                list->end = NULL;
            }
            else
            {
                nodeToDelete->prev->next = list->head;
                list->end = nodeToDelete->prev;
                list->head->prev = list->end;
            }
        
            free(nodeToDelete);
            (list->length)--;
        }

            return list->head == NULL;
