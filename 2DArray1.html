#include<stdio.h>

void main()
{
    int ar[4][2];
    int i , j;
    for(i=0;i<4;i++)
    {
        for(j=0;j<2;j++)
        {
            printf("enter element [%d] [%d]",i,j);
            scanf("%d",&ar[i][j]);
            printf("\n");
        }
    }
    //printing in matrix form
    printf("displaying all the input by the user in a matrix form");
    for(i=0;i<4;i++)
    {
        for(j=0;j<2;j++)
        {
            printf("%d",ar[i][j]);
        }
        printf("\n");
    }
}
